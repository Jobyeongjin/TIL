def blackjack(n, m, cards):
  max_total = 0
  # μ™„μ „νƒμƒ‰
  for i in range(n - 2):
    for j in range(i + 1, n - 1):
      for k in range(j + 1, n):
        total = cards[i] + cards[j] + cards[k]
        # ν„μ¬ κ°€μ¥ ν° κ°’λ³΄λ‹¤λ” ν¬κ³ , mμ€ λ„μ§€ μ•μ•„μ•Ό κ°±μ‹ 
        if max_total < total <= m:
          max_total = total
        # ν•©μ΄ mκ³Ό κ°™μΌλ©΄ νƒμƒ‰ν•  μλ―Έκ°€ μ—†μΌλ―€λ΅ μΆ…λ£
        if total == m:
          return total

  return max_total
